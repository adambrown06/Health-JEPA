\section{What This Repository Is}

\textbf{Health-JEPA} is an action-conditional JEPA for irregular clinical time series: model code under \texttt{ml/}, experiment logs under \texttt{ml/results/}, reproducibility scripts, and the paper bundle under \texttt{docs/paper/}.
Training data tensors are produced via \texttt{notebooks/aou\_extraction\_pipeline.ipynb} and loaded from \texttt{training\_data/} (not committed).

\noindent A \textbf{legacy full-stack snapshot} (Next.js UI, FastAPI, Celery, original \texttt{backend/} tree) is preserved only on git branch \texttt{archive/legacy-monorepo}.

\section{The JEPA Model (\texttt{CausalJEPA})}

\subsection{Idea in One Sentence}
The model learns joint embeddings so that \textbf{context (pre-intervention series) + intervention} predicts a \textbf{target representation (post-intervention series)} in latent space using an MSE-style loss in latent space, without raw reconstruction.

\subsection{Components (\texttt{ml/jepa\_model.py})}

\begin{longtable}{p{0.27\textwidth} p{0.68\textwidth}}
\toprule
\textbf{Piece} & \textbf{Role} \\
\midrule
\texttt{TimeSeriesEncoder} (context) &
Transformer over multivariate time-series with value + observation mask per feature and continuous time (seconds-based sinusoidal positional encoding). Produces pooled vector $s_x$ per patient. Trained with gradients. \\
\texttt{TimeSeriesEncoder} (target) &
Same architecture; weights updated only via EMA from the context encoder (\texttt{requires\_grad=False}). Produces $s_y$ as a stable target anchor. \\
\texttt{action\_embedding} &
\texttt{nn.Embedding(K, z\_dim)} mapping categorical intervention ID to dense vector. \\
\texttt{Predictor} &
MLP with AdaLN-Zero or concat conditioning: predicts $\hat{s}_y$ with the same dimensionality as $s_y$. \\
\bottomrule
\end{longtable}

\subsection{Training Loop (\texttt{ml/train\_jepa.py})}
\begin{enumerate}[leftmargin=1.5em]
  \item Forward:
  \[
  \hat{s}_y, s_y = \mathrm{model}(\mathrm{context\_x}, \mathrm{target\_y}, \mathrm{intervention\_z})
  \]
  \item Loss: $\mathrm{MSE}(\hat{s}_y, s_y)$ (or cosine variant); $s_y$ detached from loss gradients through the target encoder.
  \item Optimizer updates context encoder, action embedding, and predictor (and optional regularizers per config).
  \item EMA sync: $\theta_{\text{target}} \leftarrow m\theta_{\text{target}} + (1-m)\theta_{\text{context}}$.
\end{enumerate}

\subsection{Batch Contract}

\begin{longtable}{p{0.12\textwidth} p{0.20\textwidth} p{0.23\textwidth} p{0.40\textwidth}}
\toprule
\textbf{Symbol} & \textbf{Tensor} & \textbf{Shape (Conceptual)} & \textbf{Meaning} \\
\midrule
$x$ & \texttt{context\_x} & \texttt{(B, T\_pre, F)} & Pre-index clinical values. \\
 & \texttt{context\_mask} & \texttt{(B, T\_pre, F)} & 1 = observed, 0 = missing. \\
 & \texttt{context\_timestamps} & \texttt{(B, T\_pre)} & Seconds from window start. \\
$y$ & \texttt{target\_y} & \texttt{(B, T\_post, F)} & Post-index window. \\
 & \texttt{target\_mask}, \texttt{target\_timestamps} & Same pattern & Post-window mask and timebase. \\
$z$ & \texttt{intervention\_z} & \texttt{(B,)} \texttt{LongTensor} & Intervention ID in $0,\dots,K-1$. \\
\bottomrule
\end{longtable}
